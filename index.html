<script>

const flipbook = document.getElementById('flipbook');


// 82 single pages → 41 spreads

const totalPages = 82;

const spreads = [];


for (let i = 1; i <= totalPages; i += 2) {

  const left = String(i).padStart(2, '0');

  const right = String(i + 1).padStart(2, '0');

  spreads.push([`Portfolio_Page_${left}.jpg`, `Portfolio_Page_${right}.jpg`]);

}


const pageElements = [];

let currentSpread = 0;


// Create spreads

spreads.forEach((pair, index) => {

  const div = document.createElement('div');

  div.className = 'page';

  div.style.zIndex = spreads.length - index;


  pair.forEach(src => {

    const img = document.createElement('img');

    img.src = src;

    div.appendChild(img);

  });


  flipbook.appendChild(div);

  pageElements.push(div);

});


// Flip forward/backward with page curl effect

flipbook.addEventListener('click', (e) => {

  const rect = flipbook.getBoundingClientRect();

  const x = e.clientX - rect.left;

  let page;


  if (x > rect.width / 2) {

    // Right half → forward

    if (currentSpread < pageElements.length) {

      page = pageElements[currentSpread];

      page.classList.add('flipping');

      setTimeout(() => {

        page.classList.remove('flipping');

        page.classList.add('flipped');

        currentSpread++;

      }, 200); // short delay for curl effect

    }

  } else {

    // Left half → backward

    if (currentSpread > 0) {

      currentSpread--;

      page = pageElements[currentSpread];

      page.classList.add('flipping');

      setTimeout(() => {

        page.classList.remove('flipping');

        page.classList.remove('flipped');

      }, 200);

    }

  }

});

</script>
